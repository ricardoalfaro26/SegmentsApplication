CREATE DATABASE SegmentsApplicationDb;

CREATE TABLE RetiredCustomerSegment (
    Id INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
    PensionInstitution NVARCHAR(150) NOT NULL, -- Institución de donde recibe la pensión (AFP, ISSS, INPEP, etc.)
    PensionType NVARCHAR(50) NOT NULL, -- Pensioner | Beneficiary
    PaymentCycle NVARCHAR(50) NOT NULL, -- Monthly | Bimonthly | Quarterly | Semiannual | Annual
    PensionAmount DECIMAL(18,2) NOT NULL,
    CreatedDate DATETIME NOT NULL
);
